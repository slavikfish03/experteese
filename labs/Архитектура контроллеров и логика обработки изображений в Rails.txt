class WorkController < ApplicationController
  # Метод для отображения главной страницы рабочей области

  end


  # Метод для формирования списка тем (используется для выбора в модальном окне/форме)
  def choose_theme
    @themes = Theme.all.pluck(:name)
    respond_to :html, :js # Поддержка AJAX-запроса для отображения формы выбора
  end


  # Метод для первичного отображения выбранной темы и первого изображения
  def display_theme
    @image_data = {}
    I18n.locale = session[:current_locale]
    current_user_id = current_user.id # Предполагается наличие метода current_user из SessionsHelper


    if params[:theme] == "-----" # Если тема не выбрана
      theme = "Select theme to leave your answer"
      theme_id = 1
      values_qty = Value.all.count.round
      data = { 
        index: 0, name: 'радуга', values_qty: values_qty,
        file: 'raduga5обрез.jpg', image_id: 4,
        current_user_id: current_user_id, user_valued: false,
        common_ave_value: 0, value: 0 
      }
    else
      theme = params[:theme]
      theme_id = Theme.find_theme_id(theme) # Поиск ID темы по названию
      data = show_image(theme_id, 0) # Метод из модуля WorkImage (первое фото - индекс 0)
    end


    session[:selected_theme_id] = theme_id
    image_data(theme, data) # Вызов хелпера для упаковки данных в @image_data
  end
end

2. ApiController (app/controllers/api/api_controller.rb)
Контроллер для обработки AJAX-запросов (перелистывание изображений) без перезагрузки всей страницы 5.
module Api
  class ApiController < ApplicationController
    include WorkImage # Подключение модуля с логикой выбора изображений


    # Метод для перехода к следующему изображению
    def next_image
      current_index = params[:index].to_i
      theme_id = params[:theme_id].to_i
      length = params[:length].to_i


      # Вычисление индекса следующего изображения (с цикличностью)
      new_image_index = next_index(current_index, length)
      # Получение данных изображения из БД
      next_image_data = show_image(theme_id, new_image_index)


      respond_to do |format|
        if new_image_index.blank?
          format.html { render nothing: true, status: :unprocessable_entity }
          format.json {}
        else
          format.html { render display_theme_path, status: :successfully }
          format.json { 
            render json: { 
              new_image_index: next_image_data[:index],
              name: next_image_data[:name],
              file: next_image_data[:file],
              image_id: next_image_data[:image_id],
              user_valued: next_image_data[:user_valued],
              common_ave_value: next_image_data[:common_ave_value],
              value: next_image_data[:value],
              status: :successfully,
              notice: 'Successfully listed to next'
            } 
          }
        end
      end
    end


    # Аналогичный метод для перехода к предыдущему изображению (prev_image)
    # реализуется с использованием вспомогательного метода prev_index
  end
end

3. Модуль WorkImage (app/controllers/concerns/work_image.rb)
Общая логика, вынесенная в Concern для использования обоими контроллерами 8, 9.
module WorkImage
  extend ActiveSupport::Concern
  include WorkHelper


  # Основной метод для подготовки данных об изображении по индексу в теме
  def show_image(theme_id, image_index)
    theme_images = Image.theme_images(theme_id)
    current_user_id = current_user.id
    one_image_attr = theme_images[image_index].attributes
    image_id = one_image_attr["id"]


    # Проверка, оценивал ли уже этот пользователь данное изображение
    user_valued, value = Value.user_valued_exists(current_user_id, image_id)
    values_qty = Value.all.count.round


    if user_valued == 1
      common_ave_value = Image.find_image(image_id).ave_value
      common_ave_value = common_ave_value.blank? ? 0 : common_ave_value.round
    else
      common_ave_value = 0
    end


    { 
      index: image_index, values_qty: values_qty,
      current_user_id: current_user_id, theme_id: theme_id,
      images_arr_size: theme_images.size, image_id: image_id,
      name: one_image_attr["name"], file: one_image_attr["file"],
      user_valued: user_valued, value: value,
      common_ave_value: common_ave_value 
    }
  end


  # Логика циклического перелистывания вперед
  def next_index(index, length)
    index < length - 1 ? index + 1 : 0
  end


  # Логика циклического перелистывания назад
  def prev_index(index, length)
    index > 0 ? index - 1 : length - 1
  end
end
