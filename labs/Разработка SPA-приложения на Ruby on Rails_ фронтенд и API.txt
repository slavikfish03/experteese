  }

Скрипт перелистывания изображений (AJAX-запрос к API)
Этот код обычно размещается в основном JS-файле или внутри тега :javascript 15, 16.
$('.img-right-side').click(function(event) {
  event.stopImmediatePropagation();
  
  $.ajax({
    type: "get",
    url: "api/next_image",
    data: { 
      index: imageCurrentIndex, 
      theme_id: selectedThemeId, 
      length: themeImagesSize 
    },
    dataType: 'json'
  }).done(function (data_return) {
    // 1. Обновляем текущий индекс и данные
    imageCurrentIndex = data_return.new_image_index;
    
    // 2. Динамически меняем название и путь к файлу
    $('.image_display h2.up').append().html(data_return.name);
    var pathImage = "assets/pictures/" + data_return.file;
    $(".img-center img").attr({src: pathImage});
    
    // 3. Обновляем информацию об оценках (если есть)
    if(data_return.user_valued == 1){
      $('.left_controls span.common_value h2.common').append().html(data_return.common_ave_value);
    }
  }).fail(function (data_return) {
    console.log("error: " + data_return);
  });
});
